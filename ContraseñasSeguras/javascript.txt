// Get references to input field elements
    const emailInput = document.getElementById("email");
    const passwordInput = document.getElementById("password");
    const submitButton = document.getElementById("submitBtn");
  
    // Add event listener to each input element
    emailInput.addEventListener('focus', () => {
      emailInput.classList.add('focused');
    });
  
    emailInput.addEventListener('blur', () => {
      emailInput.classList.remove('focused');
    });
  
    passwordInput.addEventListener('focus', () => {
      passwordInput.classList.add('focused');
    });
  
    passwordInput.addEventListener('blur', () => {
      passwordInput.classList.remove('focused');
    });
  
    // Handle form submissions
    submitButton.addEventListener('click', e => {
      e.preventDefault();
      
      // Check if both email and password were entered
      if (!emailInput.value ||!passwordInput.value ) return false;

      console.log(`Email: ${emailInput.value}`);
      console.log(`Password: ${passwordInput.value}`);
      
    })
